1. Общая структура выгрузки каталога

— Категории (ключ — categories)
  — Подкатегории

— Товары (ключ — products)
  — Цвета товара
    — Размеры товара (торговые предложения)
        — Информация по складам
          — Остатки
          — Резервы
          — Поставки



2. Описание формата выгрузки каталога

Категория:
      {
        "id": 699, // int, идентификатор категории
        "name": "Ручки", // string, название категории
        "description": "Ручка – это не только популяр...", // string, описание категории
        "photo": "\/media\/images\/catalog\/00000\/34-08016681e18e7721217f34d382b420a0.jpg", // string | null, ссылка на картинку для категории. Относительно текущего домена сайта Океан Бизнес сувениров
        "subcategories": [ // array, массив подкатегорий категории, структура аналогична категории
          ...
        ]
      }

Товар:
      {
          "main_id" : 34168, // int, идентификатор главного товара модели, служит для идентификации модели товара. Совпадает с одним из идентификаторов торгового предложения текущей модели
          "name": "Поло ID.001", // string, название категории
          "brand": "B&C", // string | null, название бренда товара
          "size": "XS-XXXXL.", // string | null, размеры товара
          "info": "Рубашка поло из ткани пике плотной вязки...", // string, описание товара
          "material": "Хлопок", // string | null, материал товара
          "volume": "1440", // string, объем товара
          "memory": null, // string | null, объем памяти товара
          "weight": "0.29", // string, вес товара
          "plottings": [ // array, массив объектов доступного нанесения — если нанесения нет — массив пустой
            {
              "place": "Грудь", // string, место нанесения
              "works": [ // array of strings, массив доступных типов нанесения
                "S: Шелкография\/футболки и поло",
                "STPa: Нанесение пленкой на сложные изделия"
              ]
            },
            {...}
            ...
          ],
          "categories" : [750, 250] // array, массив идентификаторов категорий
          "colors": [ // array, массив цветов товара со вложенными размерами
            {
              "color": { // object, цвет товара
                "id": 16, // int, идентификатор цвета
                "name": "белый" // string, название цвета
              },
              "photos": [ // array, массив фотографий товара
                "\/media\/images\/catalog\/00071\/7115-e5f11afe4df4d48b69a95dbe5653f7a8.jpg", // string, относительный путь к фотографии
                ...
              ],
              "sizes": [ // массив размеров товара, даже если размеров как таковых нет (например, шкатулка), будет одна запись
                {
                  "id": 34168, // int, идентификатор торгового предложения
                  "article": "3712-1S", // артикул торгового предложения
                  "size": "S", // string | null, размер
                  "price": 559, // float, цена
                  "currency": "RUB", // string, валюта
                  "stores": { // информация по складам на складах
                    "remains": [ // массив остатков по складам
                      {
                        "store_code": "Новосибирск", // string, наименовние склада
                        "count": 79 // int, остаток
                      },
                      {
                        "store_code": "Самара",
                        "count": 170
                      }
                    ],
                    "reserves": [ // массив разервов по товару
                      {
                        "store_code": "Самара", // string, наименовние склада
                        "count": 3 // int, количество в резерве
                      }
                    ]
                  },
                  "supplies": [ // массив поставок по товару
                    {
                      "date": "2017-08-22", // sting (date Y-m-d), дата поставки
                      "count": 500 // int, количество в поставке
                    }
                  ]
                },
                ......
              ]
            }
          ]
      }

3. Описание формата выгрузки информации по складам
